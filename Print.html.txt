    // Rimuovi eventuali break precedenti (residui da versioni precedenti)
    var old = document.querySelectorAll('.page-break');
    for (var i = old.length - 1; i >= 0; i--) old[i].parentNode.removeChild(old[i]);

    var cards = Array.prototype.slice.call(
      document.querySelectorAll('.patient-card, .alt-row')
    );
    if (cards.length === 0) return;

    // Reset margini da eventuali chiamate precedenti
    cards.forEach(function(c) { c.style.marginTop = ''; });

    var MARGIN = 3; // px margin-bottom tra schede (come in CSS)

    var header     = document.querySelector('.print-header');
    var currentPos = header ? (header.offsetHeight + MARGIN) : 0;

    cards.forEach(function(card) {
      var cardH     = card.offsetHeight;
      var posInPage = currentPos % pageH;
      var remaining = pageH - posInPage;

      if (remaining >= 300) {
        // Spazio sufficiente: la scheda va qui (si spezza naturalmente se è più alta)
        currentPos += cardH + MARGIN;
      } else {
        // Meno di 300px: spingi la scheda alla pagina successiva tramite margine superiore.
        // Chrome rispetta sempre i margini — molto più affidabile di break-before su flexbox.
        var nextPageStart = Math.ceil((currentPos + 1) / pageH) * pageH;
        var gapPx = nextPageStart - currentPos;
        card.style.marginTop = gapPx + 'px';
        currentPos = nextPageStart + cardH + MARGIN;
      }
    });
  }

  // Imposta subito la larghezza di stampa per un reflow accurato, poi misura
  wrapper.style.width = printW + 'px';

  // Due frame: il primo forza il reflow del layout, il secondo garantisce
  // che tutti i calcoli di stile siano completati prima di misurare
  requestAnimationFrame(function() {
    requestAnimationFrame(function() {

      inserisciBreaks();

      wrapper.style.width      = '';
      wrapper.style.paddingTop = '52px';

      // Apre automaticamente il dialogo di stampa
      window.print();
    });
  });
})();
</script>
